CREATE PROCEDURE to_city_name ()
LANGUAGE SQL
BEGIN
 DECLARE c1 CURSOR WITH RETURN FOR 
        SELECT CITY, NAME1 AS NAME
        FROM CITY_N1N2N3
        WHERE NAME1 IS NOT NULL
        UNION 
        SELECT CITY, NAME2
        FROM CITY_N1N2N3
        WHERE NAME2 IS NOT NULL
        UNION 
        SELECT CITY, NAME3
        FROM CITY_N1N2N3
        WHERE NAME3 IS NOT NULL
        ORDER BY CITY DESC;
 OPEN c1;
END @
